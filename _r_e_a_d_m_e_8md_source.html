9 ## Library features
10 
11 * libc `<time.h>` compatible
12 * Read/write date/time `struct tm`
13 * Set/get Unix epoch UTC `time_t`
14 * Set/get time (hours, minutes, seconds)
15 * Set date and time
16 * Control `SQW` signal (disable / 1Hz / 4096Hz / 8192Hz / 32768Hz)
17 * Full RTC register access
18 
19 ## Hardware

56 ## Usage
57 
58 **Initialization**
59 
60 ```c++
61 #include <Wire.h>
62 #include <ErriezDS1307.h>
63 
64 // Create RTC object
65 ErriezDS1307 ds1307;
66 
67 void setup()
68 {
69  // Initialize I2C
70  Wire.begin();
71  Wire.setClock(100000);
72 
73  // Initialize RTC
74  while (!ds1307.begin()) {
75  Serial.println(F("Error: DS1307 not found"));
76  delay(3000);
77  }
78 
79  // Set square wave out pin
80  // SquareWaveDisable, SquareWave1Hz, SquareWave4096Hz, SquareWave8192Hz, SquareWave32768Hz
81  ds1307.setSquareWave(SquareWaveDisable);
82 }
83 ```
84 
85 **Check oscillator status at startup**
86 
87 ```c++
88 // Check oscillator status
89 if (ds1307.isOscillatorStopped()) {
90  // Error: DS1307 RTC oscillator stopped. Date/time cannot be trusted.
91  // Set new date/time before reading date/time.
92 }
93 ```
94 
95 **Set time**
96 
97 ```c++
98 // Write time to RTC
99 if (!ds1307.setTime(12, 0, 0)) {
100  // Error: RTC write failed
101 }
102 ```
103 
104 **Get time**
105 
106 ```c++
107 uint8_t hour;
108 uint8_t minute;
109 uint8_t second;
110 
111 // Read time from RTC
112 if (!rtc.getTime(&hour, &minute, &second)) {
113  // Error: RTC read failed
114 }
115 ```
116 
117 **Set date time**
118 
119 ```c++
120 // Write RTC date/time: 13:45:09 31 December 2019 2=Tuesday
121 if (!ds1307.setDateTime(13, 45, 9, 31, 12, 2019, 2) {
122  // Error: RTC write failed
123 }
124 ```
125 
126 **Write date/time struct tm**
127 
128 ```c++
129 struct tm dt = {0};
130 
131 dt.tm_hour = 12;
132 dt.tm_min = 34;
133 dt.tm_sec = 56;
134 dt.tm_mday = 29;
135 dt.tm_mon = 1; // 0=January
136 dt.tm_year = 2020-1900;
137 dt.tm_wday = 6; // 0=Sunday
138 
139 if (!ds1307.write(&dt)) {
140  // Error: RTC Read failed
141 }
142 ```
143 
144 **Read date/time struct tm**
145 
146 ```c++
147 struct tm dt = {0};
148 
149 // Read RTC date/time
150 if (!ds1307.read(&dt)) {
151  // Error: RTC read failed
152 }
153 ```
154 
155 **Read Unix Epoch UTC**
156 
157 ```c++
158 time_t t;
159 
160 // Read Unix epoch UTC from RTC
161 if (!ds1307.getEpoch(&t)) {
162  // Error: RTC read failed
163 }
164 ```
