76 bool ErriezDS1307::oscillatorEnable(bool enable)
77 {
78  uint8_t regSeconds;
79 
80  // Read seconds register
81  if (!readBuffer(DS1307_REG_SECONDS, &regSeconds, 1)) {
82  return false;
83  }
84 
85  // Set or clear HT bit
86  if (enable) {
87  // Clear to enable
88  regSeconds &= ~(1 << DS1307_SEC_CH);
89  } else {
90  // Set to disable
91  regSeconds |= (1 << DS1307_SEC_CH);
92  }
93 
94  // Write seconds register
95  return writeBuffer(DS1307_REG_SECONDS, &regSeconds, 1);
96 }

191 bool ErriezDS1307::read(struct tm *dt)
192 {
193  uint8_t buffer[7];
194 
195  // Read clock date and time registers
196  if (!readBuffer(0x00, &buffer, sizeof(buffer))) {
197  memset(dt, 0, sizeof(struct tm));
198  return false;
199  }
200 
201  // Convert BCD buffer to Decimal
202  dt->tm_sec = bcdToDec(buffer[0] & 0x7F);
203  dt->tm_min = bcdToDec(buffer[1] & 0x7F);
204  dt->tm_hour = bcdToDec(buffer[2] & 0x3f);
205  dt->tm_wday = bcdToDec(buffer[3] & 0x07) - 1;
206  dt->tm_mday = bcdToDec(buffer[4] & 0x3F);
207  dt->tm_mon = bcdToDec(buffer[5] & 0x1f) - 1;
208  dt->tm_year = 100 + bcdToDec(buffer[6]);
209 
210  return true;
211 }

221 bool ErriezDS1307::write(const struct tm *dt)
222 {
223  uint8_t buffer[7];
224 
225  // Encode date time from decimal to BCD
226  buffer[0] = decToBcd(dt->tm_sec & 0x7F);
227  buffer[1] = decToBcd(dt->tm_min & 0x7F);
228  buffer[2] = decToBcd(dt->tm_hour & 0x3F);
229  buffer[3] = decToBcd(((dt->tm_wday + 1) & 0x07));
230  buffer[4] = decToBcd(dt->tm_mday & 0x3F);
231  buffer[5] = decToBcd(((dt->tm_mon + 1) & 0x1F));
232  buffer[6] = decToBcd(dt->tm_year % 100);
233 
234  // Write BCD encoded buffer to RTC registers
235  return writeBuffer(0x00, buffer, sizeof(buffer));
236 }

381 bool ErriezDS1307::writeBuffer(uint8_t reg, void *buffer, uint8_t len)
382 {
383  // Start I2C transfer by writing the I2C address, register number and optional buffer
384  Wire.beginTransmission(DS1307_ADDR);
385  Wire.write(reg);
386  for (uint8_t i = 0; i < len; i++) {
387  Wire.write(((uint8_t *)buffer)[i]);
388  }
389  if (Wire.endTransmission(true) != 0) {
390  return false;
391  }
392  return true;
393 }

404 bool ErriezDS1307::readBuffer(uint8_t reg, void *buffer, uint8_t len)
405 {
406  // Start I2C transfer by writing the I2C address and register number
407  Wire.beginTransmission(DS1307_ADDR);
408  Wire.write(reg);
409  // Generate a repeated start, followed by a read buffer
410  if (Wire.endTransmission(false) != 0) {
411  return false;
412  }
413  Wire.requestFrom((uint8_t)DS1307_ADDR, len);
414  for (uint8_t i = 0; i < len; i++) {
415  ((uint8_t *)buffer)[i] = (uint8_t)Wire.read();
416  }
417 
418  return true;
419 }
